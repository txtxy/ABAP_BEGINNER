DATA it_flight TYPE zit_sflight_e01.
DATA wa_sflight TYPE sflight.


WRITE / 'Printout in tableorder of Database:'.

SELECT * FROM sflight
 INTO wa_sflight  "DB테이블 구조 동일
 WHERE carrid = 'JL'.
  WRITE: / wa_sflight-carrid,
           wa_sflight-connid,
           wa_sflight-fldate,
           wa_sflight-price,
           wa_sflight-currency,
           wa_sflight-planetype.
ENDSELECT.

ULINE.

SELECT * FROM sflight
 INTO TABLE it_flight    " Sorted Table 구조 복사
 WHERE carrid = 'JL'.
WRITE / 'Printout in tableorder of sorted ITAB:'.

LOOP AT it_flight INTO wa_sflight.
  WRITE: / wa_sflight-carrid,
           wa_sflight-connid,
           wa_sflight-fldate,
           wa_sflight-price,
           wa_sflight-currency,
           wa_sflight-planetype.
ENDLOOP.

*Sorted table은 이미 정렬되어 있기 떄문에,
*SORT(ABAP KEYWORD)는 사용할 수 없다.
