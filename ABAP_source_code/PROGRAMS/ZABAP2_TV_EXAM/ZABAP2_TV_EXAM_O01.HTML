MODULE get_booking_data OUTPUT.

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_bookdata
    FROM sbook
*  기준이되는 PK인 CUSTOMID 는 100번 스크린에서 sel_data로 얻어낸 정보를 이용한다.
    WHERE fldate > sy-datum AND
          customid = gs_custdata-customerid AND
          cancelled = space
    .
**********************************************************************
*  기내식을 신청했는지를 EXCP 와 REASONS 컬럼을 통해 보여줘야하기 때문에
*  REQMENU에 신청한 데이터에 관한 값을 조회해야함,
  DATA : lt_reqmenu TYPE TABLE OF ztreqmenu_e01,
         ls_reqmenu LIKE LINE OF lt_reqmenu.

  DATA : ls_reqstatus LIKE LINE OF gt_reqstatus.

*   for all entries in 주의점
  IF  gt_bookdata IS INITIAL.

    SELECT *
      INTO CORRESPONDING FIELDS OF TABLE lt_reqmenu
      FROM ztreqmenu_e01
      FOR ALL ENTRIES IN gt_bookdata
      WHERE carrid = gt_bookdata-carrid AND
            connid = gt_bookdata-connid AND
            fldate = gt_bookdata-fldate AND
            bookid = gt_bookdata-bookid.

  ENDIF.


  LOOP AT gt_bookdata INTO gs_bookdata.

    READ TABLE lt_reqmenu INTO ls_reqmenu
    WITH KEY carrid = gs_bookdata-carrid
             connid = gs_bookdata-connid
             fldate = gs_bookdata-fldate
             bookid = gs_bookdata-bookid.

    IF sy-subrc = 0 .

      READ TABLE gt_reqstatus INTO ls_reqstatus
      WITH KEY domvalue_l = ls_reqmenu-reqsts.

      IF sy-subrc = 0.
        gs_bookdata-req_statxt = ls_reqstatus-ddtext.

        IF ls_reqmenu-reqsts = 'R'.
          gs_bookdata-status = 3.

        ELSE.

          gs_bookdata-status = 1.

        ENDIF.

      ENDIF.
    ELSE.

      gs_bookdata-req_statxt = 'Not Requested'.
      gs_bookdata-status = 1.
    ENDIF.

    MODIFY gt_bookdata FROM gs_bookdata.
    CLEAR gs_bookdata.

  ENDLOOP.

ENDMODULE.

MODULE get_reqmenu OUTPUT.

  DATA : lt_mealt TYPE TABLE of smealt,
         ls_mealt LIKE LINE OF lt_mealt.

  SELECT SINGLE *
    INTO CORRESPONDING FIELDS OF ysexam2_menu_e01
    FROM ztreqmenu_e01
    WHERE carrid = gs_bookdata-carrid AND
          connid = gs_bookdata-connid AND
          fldate = gs_bookdata-fldate AND
          bookid = gs_bookdata-bookid AND
          reqsts = 'R'.

SELECT SINGLE *
  into CORRESPONDING FIELDS OF ysexam2_menu_e01
  FROM smenu
  WHERE carrid = ysexam2_menu_e01-carrid and
        menunumber = ysexam2_menu_e01-menunumber.

SELECT * into CORRESPONDING FIELDS OF TABLE lt_mealt
  FROM smealt
  where carrid = ysexam2_menu_e01-carrid and
        sprache = sy-langu.

  READ TABLE lt_mealt INTO ls_mealt
  with key mealnumber = ysexam2_menu_e01-starter.

  IF sy-subrc  = 0 .
    ysexam2_menu_e01-starter_text = ls_mealt-text.
  ENDIF.

  READ TABLE lt_mealt INTO ls_mealt
  with key mealnumber = ysexam2_menu_e01-maincourse.

  IF sy-subrc  = 0 .
    ysexam2_menu_e01-maincourse_text = ls_mealt-text.

  ENDIF.

    READ TABLE lt_mealt INTO ls_mealt
  with key mealnumber = ysexam2_menu_e01-dessert.

  IF sy-subrc  = 0 .
    ysexam2_menu_e01-dessert_text = ls_mealt-text.


  ENDIF.

ENDMODULE.
