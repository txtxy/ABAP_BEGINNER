MODULE set_book_data OUTPUT.
READ TABLE gt_book_data into gs_book_data INDEX gv_sel_row.
DATA : lv_menureq TYPE ztreqmenu_e01-reqsts.

    SELECT  *
      INTO CORRESPONDING FIELDS OF gs_sbook
      FROM sbook
      WHERE customid = gs_book_data-id AND
            class IN gr_class AND
            cancelled <> 'X' and
            fldate > to_day.
            .

      SELECT SINGLE reqsts
        into lv_menureq
        from ztreqmenu_e01
        where bookid = gs_sbook-bookid.

PERFORM GET_DOMAIN_DATA USING 'ZDREQSTS_E01'
                               lv_menureq
                               gs_sbook-REasons.


      IF lv_menureq = 'R'.

        gs_sbook-excp = '3'.
       else .
         gs_sbook-excp = '1'.
      ENDIF.
APPEND gs_sbook TO gt_sbook.
ENDSELECT.
