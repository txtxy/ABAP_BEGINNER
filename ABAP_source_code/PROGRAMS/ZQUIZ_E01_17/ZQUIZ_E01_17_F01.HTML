FORM get_data .

* 평가항목 점수를 받아올 LOCAL STRUCTURE 선언.

  DATA : BEGIN OF ls_evl ,
           evl1 TYPE zteval_e00-evl1,
           evl2 TYPE zteval_e00-evl2,
           evl3 TYPE zteval_e00-evl3,
         END OF ls_evl.

* LIKE 검색을 위한 와일드 카드 부여.

  pa_cnm = '%' && pa_cnm && '%'.

* 기본 데이터 조회
* 반복문으로 계산값을 동시에 부여할 계획임.

  SELECT * INTO CORRESPONDING FIELDS OF gs_data
*                                       1번 DB  scustom  기준
    FROM scustom AS c
*                                       2번 DB SBOOK
     JOIN sbook AS b
*                                       CUSTOMID 가 FK조건
    ON c~id = b~customid
*                                       3번 DB SCARR
    JOIN scarr AS a
*                                       CARRID 가 FK 조건
    ON b~carrid = a~carrid
*                                       4번 DB  zteval_e00
    JOIN zteval_e00 AS z
*                                       PK인 항목을 조건으로 작성.
    ON b~customid = z~customid AND
       b~bookid = z~bookid AND
       b~connid = z~connid AND
       b~fldate = z~fldate

    WHERE c~id IN so_cid AND         " S-O 조건 입력
          b~fldate IN so_fld AND     " S-O 조건 입력
          b~custtype IN so_bp AND    " 실질 PA지만 미입력시 전체 조회를 위해 S-O로 조건
          c~name LIKE pa_cnm .       " NAME의 LIKE 조회

*   기본 데이터 1개 ROW조회
*--------------------------------------------------------------------*

*  GS_DATA 미입력 사항 입력

* Average 계산을 위한 기초 데이터 조회.

    SELECT SINGLE evl1 evl2 evl3 INTO ls_evl  " 평점 STRUCTURE로 받아오기
      FROM zteval_e00
      WHERE carrid = gs_data-carrid AND       " 다중 식별자를 이용하여 UNIQUE RECORD 조회.
            connid = gs_data-connid AND
            fldate = gs_data-fldate AND
            customid = gs_data-id .

* 평점 평균 값 적용.
    gs_data-average = ( ls_evl-evl1 + ls_evl-evl2 + ls_evl-evl3  ) DIV 3.

* 도메인 이름 값 획득.

    PERFORM get_domain_name USING 'S_CLASS'
                                   gs_data-class
                                   gs_data-class_name.

* 1개 ROW의 값을 GT_DATA ITAB으로 차근차근 입력하기.
    APPEND gs_data TO gt_data.

  ENDSELECT.




ENDFORM.

      values_tab      = lt_value
    EXCEPTIONS
      no_values_found = 1
      OTHERS          = 2.

  IF sy-subrc <> 0.

  ENDIF.

    WITH KEY domvalue_l = pv_value.
