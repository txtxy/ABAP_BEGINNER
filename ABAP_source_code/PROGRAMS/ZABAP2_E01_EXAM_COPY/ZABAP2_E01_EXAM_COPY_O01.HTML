MODULE set_menu_data OUTPUT.





  SELECT * FROM smenu
    INTO CORRESPONDING FIELDS OF gs_menu
    WHERE carrid = gs_sbook-carrid .

    SELECT SINGLE text
      INTO gs_menu-appi
      FROM smealt
      WHERE carrid = gs_sbook-carrid AND
            mealnumber = gs_menu-starter.

    SELECT SINGLE text
      INTO gs_menu-main
      FROM smealt
      WHERE carrid = gs_sbook-carrid AND
            mealnumber = gs_menu-maincourse.


    SELECT SINGLE text
      INTO gs_menu-desert
      FROM smealt
      WHERE carrid = gs_sbook-carrid AND
            mealnumber = gs_menu-dessert.

    APPEND gs_menu TO gt_menu.
  ENDSELECT.






ENDMODULE.
