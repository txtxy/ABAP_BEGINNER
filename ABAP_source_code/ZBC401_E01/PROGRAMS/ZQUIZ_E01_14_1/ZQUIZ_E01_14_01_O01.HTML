MODULE get_top5 OUTPUT.
  DATA :
    gv_fname TYPE string,
    gv_index TYPE num2.

  FIELD-SYMBOLS : <fs> TYPE any.


  SORT gt_data BY days DESCENDING.



  LOOP AT gt_data INTO gs_data FROM 1 TO 5.

gv_index = sy-tabix.

    gv_fname = 'GS_EARLY-CUSTOMID' && gv_index.
       ASSIGN (gv_fname) TO <fs>.
       <fs> = gs_data-customid.

    gv_fname = 'GS_EARLY-NAME' && gv_index.
        ASSIGN (gv_fname) TO <fs>.
       <fs> = gs_data-name.

    gv_fname = 'GS_EARLY-DAYS' && gv_index.
        ASSIGN (gv_fname) TO <fs>.
       <fs> = gs_data-days.

ENDLOOP.
ENDMODULE.
