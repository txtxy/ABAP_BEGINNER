FORM get_basic_data .
  DATA : lv_year TYPE c LENGTH 5.
  lv_year = pa_year && '%'.

  SELECT a~carrid a~carrname b~fldate  b~paymentsum AS paysum
    INTO CORRESPONDING FIELDS OF TABLE gt_pay_group
    FROM scarr AS a
    JOIN sflight AS b
    ON a~carrid = b~carrid
    WHERE b~fldate LIKE lv_year AND
    b~paymentsum <> 0 .

ENDFORM.

FORM mothly_grouping .
  DATA :
    lv_month TYPE string,
    lv_cnt   TYPE num2 VALUE '01',
    lv_sum   TYPE sflight-paymentsum.

  SORT gt_pay_group BY carrid fldate ASCENDING.

  LOOP AT gt_scarr INTO wa_scarr.


    WHILE lv_cnt < 13.
      LOOP AT gt_pay_group INTO gs_pay_group .
        IF gs_pay_group-carrid > wa_scarr-carrid AND
           gs_pay_group-fldate(6) > pa_year(4) && lv_cnt.
          EXIT.

        ELSE.
          IF gs_pay_group-carrid = wa_scarr-carrid AND
             gs_pay_group-fldate(6) = pa_year(4) && lv_cnt.

            lv_sum = lv_sum + gs_pay_group-paysum.


          ENDIF.
        ENDIF.

      ENDLOOP.

      lv_month = 'GS_PAY_SUM-MON' && lv_cnt .

      ASSIGN (lv_month) TO <fv_month>.
      <fv_month> = lv_sum.

      CLEAR : lv_sum.

      ADD 1 TO lv_cnt.
    ENDWHILE.

    gs_pay_sum-carrid = wa_scarr-carrid.
    gs_pay_sum-carrname = wa_scarr-carrname.


    APPEND gs_pay_sum TO gt_pay_sum.
    CLEAR : gs_pay_sum,lv_cnt.


  ENDLOOP.


ENDFORM.
