MODULE GET_DATA INPUT.

  SELECT SINGLE *
    INTO CORRESPONDING FIELDS OF WA_CONNECT
    FROM SPFLI
    WHERE CARRID = SDYN_CONN-CARRID
      AND CONNID = SDYN_CONN-CONNID.

  IF SY-SUBRC <> 0.
    CLEAR WA_CONNECT.
    MESSAGE 'Sorry, Data Not Found' TYPE 'E'.
  ENDIF.
ENDMODULE.
