MODULE status_0100 OUTPUT.

  IF maintain_flights = 'X'.
    SET PF-STATUS 'STATUS_0100'.
  ELSE .
    SET PF-STATUS 'STATUS_0100' EXCLUDING 'SAVE' .
  ENDIF.

  CASE 'X'.
    WHEN view.
      SET TITLEBAR 'TITLE_100' WITH 'VIEW'(vie).
    WHEN maintain_flights.
      SET TITLEBAR 'TITLE_100' WITH 'Maintain Flights'(fli).
    WHEN maintain_bookings.
      SET TITLEBAR 'TITLE_100' WITH 'Maintain Bookings'(boo).
  ENDCASE.
ENDMODULE.

MODULE modify_screen OUTPUT.

  CASE 'X'.
    WHEN maintain_flights .
      LOOP AT SCREEN.

        IF screen-name = 'SDYN_CONN-PLANETYPE'.
          screen-input = 1.
          MODIFY SCREEN.
        ENDIF.
      ENDLOOP.
  ENDCASE.



ENDMODULE.

    WHEN 'FC2'.
      gv_dynnr = '0120'.
    WHEN 'FC3'.
      gv_dynnr = '0130'.

  ENDCASE.

MODULE get_spfli OUTPUT.
  SELECT SINGLE *
    INTO CORRESPONDING FIELDS OF sdyn_conn
    FROM spfli
    WHERE carrid = sdyn_conn-carrid AND
          connid = sdyn_conn-connid .
  MOVE-CORRESPONDING sdyn_conn TO sdyn_conn.
ENDMODULE.
