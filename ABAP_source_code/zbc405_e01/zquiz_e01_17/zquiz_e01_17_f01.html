FORM get_data .

* #### ### ### LOCAL STRUCTURE ##.

  DATA : BEGIN OF ls_evl ,
           evl1 TYPE zteval_e00-evl1,
           evl2 TYPE zteval_e00-evl2,
           evl3 TYPE zteval_e00-evl3,
         END OF ls_evl.

* LIKE ### ## ### ## ##.

  pa_cnm = '%' && pa_cnm && '%'.

* ## ### ##
* ##### #### ### ### ###.

  SELECT * INTO CORRESPONDING FIELDS OF gs_data
*                                       1# DB  scustom  ##
    FROM scustom AS c
*                                       2# DB SBOOK
     JOIN sbook AS b
*                                       CUSTOMID # FK##
    ON c~id = b~customid
*                                       3# DB SCARR
    JOIN scarr AS a
*                                       CARRID # FK ##
    ON b~carrid = a~carrid
*                                       4# DB  zteval_e00
    JOIN zteval_e00 AS z
*                                       PK# ### #### ##.
    ON b~customid = z~customid AND
       b~bookid = z~bookid AND
       b~connid = z~connid AND
       b~fldate = z~fldate

    WHERE c~id IN so_cid AND         " S-O ## ##
          b~fldate IN so_fld AND     " S-O ## ##
          b~custtype IN so_bp AND    " ## PA## #### ## ### ## S-O# ##
          c~name LIKE pa_cnm .       " NAME# LIKE ##

*   ## ### 1# ROW##
*--------------------------------------------------------------------*

*  GS_DATA ### ## ##

* Average ### ## ## ### ##.

    SELECT SINGLE evl1 evl2 evl3 INTO ls_evl  " ## STRUCTURE# ####
