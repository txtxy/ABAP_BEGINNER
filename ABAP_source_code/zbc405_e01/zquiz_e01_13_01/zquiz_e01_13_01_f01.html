FORM get_data.

* ## SBOOK ### ##
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_book
    FROM sbook
    WHERE carrid IN so_car AND
          connid IN so_con AND
          fldate IN so_fld.

* DAYs # ## # EXCP ##.
  LOOP AT gt_book INTO gs_book.

    gs_book-days = ( gs_book-fldate - gs_book-order_date ).

    IF gs_book-days < 50.            " orderdate - fldate# 50### #

      gs_book-excp = '1'.           " #####

    ELSEIF gs_book-days >= 50 AND
           gs_book-days < 100.        " orderdate - fldate# 50## 100### #

      gs_book-excp = '2'.            " #####

    ELSE.

      gs_book-excp = '3'.            " ## ## #####

    ENDIF.

    MODIFY gt_book FROM gs_book.     " ## # ##.

  ENDLOOP.
