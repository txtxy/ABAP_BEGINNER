MODULE get_store_data OUTPUT.
  DATA :
         BEGIN OF ls_code,
           store_code TYPE ztstore_cl5-store_code,
         END OF ls_code,
         ls_store LIKE LINE OF gt_store.

  DATA : lt_code  like TABLE OF ls_code.


  CASE gv_cate .

    WHEN 'KOREAN'.
      SELECT store_code INTO CORRESPONDING FIELDS OF TABLE lt_code
        FROM ztmenu_cl5
        WHERE category = 'KOREAN'.

    WHEN 'ITALY'.
      SELECT store_code INTO CORRESPONDING FIELDS OF TABLE lt_code
    FROM ztmenu_cl5
    WHERE category = 'ITALY'.

    WHEN 'CHINA'.
      SELECT store_code INTO CORRESPONDING FIELDS OF TABLE lt_code
FROM ztmenu_cl5
WHERE category = 'CHINA'.

    WHEN 'JAP'.
      SELECT store_code INTO CORRESPONDING FIELDS OF TABLE lt_code
FROM ztmenu_cl5
WHERE category = 'JAP'.

    WHEN OTHERS.
      gv_cate =''.

  ENDCASE.

  IF gv_cate <> ''.

    LOOP AT lt_code INTO ls_code.

      SELECT SINGLE *
        INTO CORRESPONDING FIELDS OF ls_store
        FROM ztstore_cl5
        WHERE store_code = ls_code-store_code.

      APPEND ls_store TO gt_store.

    ENDLOOP.

  ELSE.
    SELECT * FROM ztstore_cl5
    INTO CORRESPONDING FIELDS OF TABLE gt_store.


  ENDIF.






ENDMODULE.
