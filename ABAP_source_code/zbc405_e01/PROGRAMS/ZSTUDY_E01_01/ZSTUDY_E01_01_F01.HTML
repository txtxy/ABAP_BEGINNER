FORM check_id .

  DATA : lv_check TYPE  i.

  SELECT COUNT(*)
    INTO lv_check
    FROM ztuser_cl5
    WHERE user_code = ztuser_cl5-user_code AND
          password = ztuser_cl5-password.
  IF sy-subrc = 0 .
    CALL SCREEN 200.
  ELSE.
    MESSAGE TEXT-fal TYPE 'I'.
    CLEAR : lv_check , ztuser_cl5.
    LEAVE TO SCREEN 100.
  ENDIF.


ENDFORM.
