REPORT ZQUIZ_E01_02 MESSAGE-ID zmessage_e01.
* 스크린 생성
PARAMETERS pa_num TYPE i.

*구구단 결과 변수 설정
DATA gv_result like pa_num.

*조건문 시작
IF pa_num <= 1.
* 입력값이 1 이하일때
*  WRITE:/ '2이상을 입력해주세요.'.
*  MESSAGE i000(zmessage_e01).
  MESSAGE '2이상을 입력해주세요.' type 'I'.

ELSEIF pa_num > 9.
* 입력값이 10 이상일 때.
*  WRITE:/ '9이하를 입력하여 주세요.'.
*  MESSAGE i001.
  MESSAGE i001 DISPLAY LIKE 'E'.

else.
* 정상 값 입력시.
  WRITE:/ '구구단입니다'.
  DO 9 TIMES.
* 9번 반복
    gv_result = pa_num * sy-index.
* 구구단 결과 값 입력
    WRITE:/ pa_num, '*', sy-index CENTERED, ' = ',gv_result LEFT-JUSTIFIED.
* 결과 출력
  ENDDO.
* 반복 종료
ENDIF.
*조건문 종료
