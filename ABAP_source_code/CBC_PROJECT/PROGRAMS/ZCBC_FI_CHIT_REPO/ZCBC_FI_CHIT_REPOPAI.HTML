MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BASIC' OR 'ADDIT'.
      my_tab-activetab = ok_code.

    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'EXECUTE'.
      PERFORM get_seldata.
    when 'REF'.
      PERFORM clean_sel.
      clear : gt_disph, gt_dispi.
      PERFORM refresh_head.
      PERFORM refresh_item.

    WHEN 'CREATE'.
      CALL TRANSACTION 'ZCBC_FI_CHIT'.

  ENDCASE.
ENDMODULE.
