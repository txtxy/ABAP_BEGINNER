FORM get_seldata .

  DATA : ls_disph LIKE LINE OF gt_chith.

  IF so_bp IS INITIAL.

    SELECT * FROM ztc_fi_chit_h
      INTO CORRESPONDING FIELDS OF TABLE gt_chith
      WHERE  fi_year =  pa_year AND
             chit_doc_num IN  so_chit AND
              post_day IN  so_post AND
               prf_day IN  so_prf  AND
              co_cen IN   so_cocen AND
               chit_type IN so_chtp AND
               doc_num IN  so_doc  .

  ELSE.

    SELECT * FROM ztc_fi_chit_h
      INTO CORRESPONDING FIELDS OF TABLE @gt_chith
      WHERE fi_year    =  @pa_year AND
            prf_day   IN  @so_prf  AND
            chit_type IN  @so_chtp AND
            doc_num   IN  @so_doc AND
           chit_doc_num IN ( SELECT chit_num FROM ztc_fi_chit_i
                               WHERE  chit_num  IN  @so_chit AND
                                      post_day  IN  @so_post AND
                                      bp_for    IN  @so_bp ) .


  ENDIF.
  MOVE-CORRESPONDING gt_chith TO gt_disph .
  SORT gt_disph DESCENDING by chit_doc_num.

  PERFORM add_disp_data.



ENDFORM.

*&---------------------------------------------------------------------*
*& text

*& <--  p2        text
*&---------------------------------------------------------------------*
FORM add_disp_data .
  DATA :ls_emp LIKE LINE OF gt_emp,
        ls_ctp LIKE LINE OF gt_ctp.

  LOOP AT gt_disph INTO gs_disph.

    READ TABLE gt_emp INTO ls_emp WITH KEY emp_id = gs_disph-ernam .
    MOVE ls_emp-name TO gs_disph-ername.
    CLEAR ls_emp.

    READ TABLE gt_emp INTO ls_emp WITH KEY emp_id = gs_disph-aenam .
    MOVE ls_emp-name TO gs_disph-aename.
    clear : ls_emp.

    READ TABLE gt_ctp INTO ls_ctp WITH KEY chit_tp_id = gs_disph-chit_type .
    MOVE ls_ctp-chit_tp_name TO gs_disph-chit_tp_name.
    clear : ls_ctp.


    MODIFY gt_disph FROM gs_disph.

  ENDLOOP.

ENDFORM.
