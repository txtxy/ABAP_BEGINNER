*&---------------------------------------------------------------------*
FORM get_item_data .
  CLEAR : gt_mm_i,gs_mm_i.
  DATA : ls_stf LIKE LINE OF gt_stf,
         ls_bp  LIKE LINE OF gt_bp.



  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_mm_i
    FROM ztc_mm_ing_i
    WHERE conf_doc_num = gs_mm_h-conf_doc_num.

  SORT gt_mm_i ASCENDING BY conf_doc_num bp_id.

  LOOP AT gt_mm_i INTO gs_mm_i.

    gs_mm_i-inv_unit = 'EA'.
    gs_mm_i-req_unit = 'EA'.
    gs_mm_i-currency = 'KRW'.
    gs_mm_i-total_currency = 'KRW'.

    READ TABLE gt_stf INTO ls_stf WITH KEY stuff_id = gs_mm_i-stuff_id .
    MOVE ls_stf-stf_name TO gs_mm_i-stuff_name.

    READ TABLE gt_bp INTO ls_bp WITH KEY bp_id = gs_mm_i-bp_id.
    MOVE ls_bp-bp_name TO gs_mm_i-bp_name.

    SELECT SINGLE inv_qty
      INTO gs_mm_i-inv_qty
      FROM ztc_wm_inv
      WHERE inv_id = gs_mm_i-stuff_id .



    gs_mm_i-total_price = gs_mm_i-price * gs_mm_i-req_qty.

    MODIFY gt_mm_i FROM gs_mm_i.
    CLEAR gs_mm_i.
  ENDLOOP.


ENDFORM.
