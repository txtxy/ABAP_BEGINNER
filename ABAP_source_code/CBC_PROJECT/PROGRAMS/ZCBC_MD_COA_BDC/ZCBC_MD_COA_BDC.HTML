  DATA p_filename  LIKE rlgrap-filename.

    EXPORTING                                        " 펑션 요소
      filename                = p_file                   "불러올 파일의 절대경로와 이름

      i_end_col               = lo_end_col               "컬럼 / 세로줄 끝자리
      i_end_row               = lo_end_row               "몇줄까지 불러올 것인지?

  SORT lo_itab BY row col.

  DATA :  ls_wwwdata TYPE wwwdatatab,
          lv_file    TYPE rlgrap-filename.

  SELECT SINGLE * INTO CORRESPONDING FIELDS OF ls_wwwdata
    FROM wwwdata
    WHERE objid = 'ZCBC_MD_COA_FILE'.  "SMW0 에서 저장해둔 엑셀 파일

  CHECK pv_savepath IS NOT INITIAL.

  "타입 변환을 위해 아래와 같이 값을 전달해준다.
