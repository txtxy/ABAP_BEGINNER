FORM get_dept .

  CASE pa_del.
    WHEN 'X'.
      SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM ztc_md_dept AS a
    JOIN ztc_md_dept_t  AS b
    ON a~dept_id = b~dept_id AND
    b~langu = sy-langu
    WHERE a~dept_id IN so_id AND
          b~dept_name IN so_name.

    WHEN space.

      SELECT a~dept_id AS dept_id b~dept_name AS dept_name a~del_flag as del_flag
             b~langu as langu
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM ztc_md_dept AS a
    JOIN ztc_md_dept_t  AS b
    ON a~dept_id = b~dept_id AND
    b~langu = sy-langu
    WHERE a~dept_id IN so_id AND
          b~dept_name IN so_name AND
        a~del_flag = space.
