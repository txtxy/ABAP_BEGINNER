  gs_chiti-right_acc = '0050010101'.   " 재료 원가
  gs_chiti-cred_curr = 'KRW'.

        '   매입 확정의 건'.

    INTO ( gs_chith-ernam, gs_chith-co_cen )
     FROM ztc_md_emp
    WHERE login_id = sy-uname.

  gs_chith-erzet = sy-timlo.
  gs_chith-post_day = gs_chith-erdat.
*    원청문서 생성일
  gs_chith-prf_day = gs_chith-erdat.
  .
  CALL FUNCTION 'NUMBER_GET_NEXT'
    EXPORTING
      nr_range_nr             = '07'
      object                  = 'ZCBC_RANG'
*     QUANTITY                = '1'
*     SUBOBJECT               = ' '
*     TOYEAR                  = sy-datum(4)
*     IGNORE_BUFFER           = ' '
    IMPORTING
      number                  = Pv_chit_num
*     QUANTITY                =
*     RETURNCODE              =
    EXCEPTIONS
      interval_not_found      = 1
      number_range_not_intern = 2
      object_not_found        = 3
      quantity_is_0           = 4
      quantity_is_not_1       = 5

      buffer_overflow         = 7
